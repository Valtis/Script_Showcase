e = epsilon

TOPLEVEL ::= STATICS FUNCTIONS
STATICS ::= (statics IDENTIFIER_LIST) | e
IDENTIFIER_LIST ::= identifier IDENTIFIER_LIST | e
FUNCTIONS ::= FUNCTION FUNCTIONS | e
FUNCTION ::= (fn identifier (IDENTIFIER_LIST) (STATEMENTS))
STATEMENTS ::= STATEMENT STATEMENTS  
STATEMENT ::= (EXPRESSION) | IF_STATEMENT | (ASSIGNMENT)
EXPRESSION ::=  ARITHMETIC_EXPRESSION | NATIVE_INVOKATION | FUNCTION_CALL | LITERAL_OR_IDENTIFIER
ARITHMETIC_EXPRESSION ::= (+|-|*|/ ARGUMENT_LIST)
IF_STATEMENT = (if (COMPARISON_OPERATOR EXPRESSION EXPRESSION) (STATEMENT) (STATEMENT)|e )
NATIVE_INVOKATION ::= (invokenative string|identifier ARGUMENT_LIST)
FUNCTION_CALL ::= (identifier ARGUMENT_LIST)
LITERAL ::= identifier|number|string
ARGUMENT_LIST ::= EXPRESSION ARGUMENT_LIST | e
ASSIGNMENT ::= setvalue identifier identifier|number|string|


Identifier may start with lower or uppercase letter a - z or underscore. May contain the previous and numbers.
String stars and ends with '"'-character. Escape sequences \n, \t, \\ and \" 

Reserved words
