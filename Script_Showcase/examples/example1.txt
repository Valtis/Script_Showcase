statics
  string printer as Printer
endstatics

# calculates (1 + 2) * 3 and prints the result. 
# Then calculates 6 / (1 + 2) and prints the result
function example
  locals
  endlocals
    
  InvokeManaged first
  InvokeManaged second
  
endfunction

function first
  locals
  endlocals
    
  PushInteger 1 
  PushInteger 2
  IntegerAdd
  PushInteger 3
  IntegerMultiply
  
  InvokeManaged printValue
endfunction

# uses locals for showcasing purposes; not actually required for this
function second
  locals
    additionResult
  endlocals
  PushInteger 1
  PushInteger 2
  IntegerAdd
  StoreLocal additionResult
  PushInteger 6
  LoadLocal additionResult
  IntegerDivide

  InvokeManaged printValue
  
endfunction

function printValue
  locals
  endlocals
  LoadStatic Printer
  InvokeNative
endfunction
